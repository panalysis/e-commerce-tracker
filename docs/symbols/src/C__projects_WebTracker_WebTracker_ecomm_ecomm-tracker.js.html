 77

 90    this.domain = domain;
 91    this.docpath = dp;

304 /**

309 WebTracker.prototype._getCookie = function(strParam) {
310    var dcookies = document.cookie.split(";"),
311        nvp = [],
312        val = "", strParam = strParam || "";
313    
314    for(var i = 0, count = dcookies.length; i < count; i++) {
315       nvp = dcookies[i].split("=");
316       if( nvp[0].toLowerCase().trim()==strParam.toLowerCase() ) {
317          for(var j = 1, npl = nvp.length; j < npl; j++) {
318             if( j>1 )
319                val += "=";

324       };

326

329 /**
330 * Retrieves the a query parameter from the document hash
331 * @param {String} strParam The name of the parameter to find.
332 * @return {String} The query parameter value if found.
333 */
334 WebTracker.prototype._getHashParam = function(strParam) {
335    if( !strParam )
336       return "";
337    
338    var dochash = document.location.hash.substring(1),
339        hashparams = dochash.split("&"),
340        paramlist = {}, np = [], 
341        hp = "", strParam = strParam || "";
342    
343    for(var i = 0, l = hashparams.length; i < l; i++) {
344       np = hashparams[i].split("=");
345       if( np.length==2 ) {
346          paramlist[np[0].trim()] = np[1].trim();
347       }
348       if( !!strParam && np[0].toLowerCase().trim()==strParam.toLowerCase() ) {
349          hp = np[1].trim();
350          break;
351       }
352    }
353    
354    return hp;
355 };
