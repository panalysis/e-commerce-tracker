  • checkBannerClick()
    Performs a check for indication of banner click (indicated by well-known hash param 'ac') and if found, records event. Should be called during init. Event details: category = "Promo Click".
  • checkForNoSearchResults()
    Performs a check for zero search results. (Should only execute on a search results page.) If a zero search result is found then make an appropriate call to _trackProductSearchNoResults.
  • init()

   {

      wto.checkForNoSearchResults();
   }

  • tagAddToCart()
    Tags elements for add to cart tracking. Appropriate elements should be identified and event handlers added that include call to _trackProductAdd. 
       WebTracker.prototype.tagAddToCart = function()
   {
      var wto = this,
          action = "Not Specified", 
          productInfo = [];

      if( !!$("h1#product-category").length )
         action = $("h1#product-category").text();

      $("#add-to-cart-btn").click(function(ev) {
         var data = { Action:action, Label:"", Value:1 };
         productInfo[0] = $("span.product-id").text();
         productInfo[1] = $("span.product-name").text();
         data.Label = productInfo.join("-");
      
         wto._trackProductAdd(data);
      });
   };
